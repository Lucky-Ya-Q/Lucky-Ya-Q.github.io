# Promise 封装 fs 读取文件

const fs = require('fs')

function myReadFile(path) {
    return new Promise((resolve, reject) => {
        fs.readFile(path, (err, data) => {
            if (err) {
                reject(err)
            } else {
                resolve(data.toString())
            }
        })
    })
}

myReadFile('./package.json').then(value => {
    console.log(value)
}, reason => {
    console.log(reason)
})
